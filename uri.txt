    if(A == 0)
    {
        printf("Impossivel calcular\n");
    }
    else if(c<0)
    {
        printf("Impossivel calcular\n");

#include<iostream>
using namespace std;
int A[3],temp[3],i;
int main()
{
    for(int i = 0;i<3;i++)
    {
        cin>>A[i];
    }
    for(int i = 0;i<3;i++)
    {
        temp[i] = A[i];
    }
    for(int i=0;i<3;i++)
    {
        for(int j = i+1;j<3;j++)
        {
            if(A[i]>A[j])
            {
                int t = A[i];
                A[i] = A[j];
                A[j] = t;
            }
        }
    }
    for(int i = 0;i<3;i++)
    {
        cout<<A[i]<<endl;
    }
    cout<<endl;
    for(int i = 0;i<3;i++)
    {
        cout<<temp[i]<<endl;
    }
}
